cNoti.addEventListener('characteristicvaluechanged', (e) => { let data = new Uint8Array(e.target.value.buffer); // Si la cadena empieza con 0x12 y tiene el largo correcto if (data[0] === 0x12 && data.length >= 15) { // --- PESO REVISADO --- // Si el peso es 110.5, y el byte 11 es 0x36 (54), // la báscula está enviando unidades de 2kg + un ajuste. // Pero para captar 110.3, necesitamos ver si el byte 12 (0x01) influye. let pesoLimpio = ((data[11] * 2) + (data[12] * 0.1) + 2.4).toFixed(1); document.getElementById('peso').innerText = pesoLimpio; // --- IMPEDANCIA REVISADA (Bytes 13 y 14) --- let ohms = (data[13] << 8) | data[14]; // Si el valor es coherente (más de 100 ohms y menos de 3000) if (ohms > 100 && ohms < 3000) { document.getElementById('imp').innerText = ohms; // Log para auditoría en el iPad console.log("Lectura detectada - Peso: " + pesoLimpio + " Imp: " + ohms); } } });
